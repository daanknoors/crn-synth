 53class NumericGeneralizationMech(BaseGeneralizationMech):
 54    """Generalization mechanism for numeric data"""
 55
 56    def __init__(
 57        self,
 58        column,
 59        bins,
 60        bounds,
 61        epsilon=None,
 62        inverse="truncated_normal",
 63        random_state=None,
 64    ):
 65        if inverse == "uniform" and epsilon is not None:
 66            print(
 67                "No epsilon needed for inverse using uniform distribution. Epsilon will be set to None."
 68            )
 69            epsilon = None
 70
 71        super().__init__(
 72            column=column,
 73            epsilon=epsilon,
 74            bins=bins,
 75            bounds=bounds,
 76            inverse=inverse,
 77            random_state=random_state,
 78        )
 79
 80    def fit(self, data):
 81        """Fit the generalization mechanism to the data."""
 82        self._check_params()
 83        self._check_data(data)
 84        self.bin_edges_ = self._get_bin_edges(bins=self.bins, bounds=self.bounds)
 85
 86        # save dp mean and std for inverse transformation prior to generalization
 87        if self.inverse == "truncated_normal":
 88            # only compute based on non-nan values
 89            array = self._get_array(data)
 90
 91            # split epsilon budget for mean and std
 92            eps_param = self.epsilon / 2
 93
 94            # compute dp mean and std
 95            self.dp_mean_ = dp_mean(
 96                array=array,
 97                epsilon=eps_param,
 98                bounds=self.bounds,
 99                random_state=self.random_state,
100            )
101            self.dp_std_ = dp_std(
102                array=array,
103                epsilon=eps_param,
104                bounds=self.bounds,
105                random_state=self.random_state,
106            )
107
108        return self
109
110    def transform(self, data):
111        """Transform the data using the generalization mechanism."""
112        data = data.copy()
113
114        # only transform non-nan values
115        array = self._get_array(data)
116
117        # clip values outside of bounds
118        array = np.clip(array, self.bounds[0], self.bounds[1])
119
120        # bin the data
121        binned_data = np.digitize(array, self.bin_edges_[1:], right=True)
122
123        # replace non-nan values with binned values
124        data.loc[~data[self.column].isna(), self.column] = binned_data
125        return data
126
127    def inverse_transform(self, data):
128        """Inverse transform the data using the generalization mechanism."""
129        data = data.copy()
130
131        # only inverse transform non-nan values
132        array = self._get_array(data)
133
134        # ensure integer bins as nan-values may have converted bins to float
135        array = np.int_(array)
136
137        # get bounds of bins
138        low_bound = self.bin_edges_[array]
139        up_bound = self.bin_edges_[array + 1]
140
141        # sample from uniform distribution
142        if self.inverse == "uniform":
143            if self.dtype_ == "float":
144                sampled_values = np.random.uniform(low=low_bound, high=up_bound)
145            elif self.dtype_ == "int":
146                sampled_values = np.random.randint(low=low_bound, high=up_bound)
147            else:
148                raise ValueError("Data type must be either 'int' or 'float'.")
149
150        # sample from truncated normal distribution
151        elif self.inverse == "truncated_normal":
152            low = (low_bound - self.dp_mean_) / self.dp_std_
153            upp = (up_bound - self.dp_mean_) / self.dp_std_
154            X = truncnorm(a=low, b=upp, loc=self.dp_mean_, scale=self.dp_std_)
155
156            sampled_values = X.rvs(random_state=self.random_state)
157            if self.dtype_ == "int":
158                sampled_values = np.around(sampled_values, decimals=0).astype(int)
159
160        # replace non-nan values with sampled values
161        data.loc[~data[self.column].isna(), self.column] = sampled_values
162        return data
163
164    def _check_params(self):
165        """Check the parameters of the generalization mechanism."""
166        if self.inverse not in ["truncated_normal", "uniform"]:
167            raise ValueError(
168                "Inverse transformation must be either 'truncated_normal' or 'uniform'."
169            )
170
171        return super()._check_params()
172
173    def _check_data(self, data):
174        """Check the data for the generalization mechanism."""
175        # check if data is numeric
176        if not np.issubdtype(data[self.column].dtype, np.number):
177            raise ValueError("Data must be numeric.")
178
179        return super()._check_data(data)
180
181    def _get_array(self, data):
182        """Get the array for generalization."""
183        # only transform non-nan values
184        return data.loc[~data[self.column].isna(), self.column]
185
186    def _get_bin_edges(self, bins, bounds):
187        """Get the bin edges based on bounds and number of bins."""
188        # get equal-width bins when bins is integer
189        if isinstance(bins, int):
190            # take uniform bins
191            bin_edges = np.linspace(bounds[0], bounds[1], bins + 1)
192
193            # round to 2 decimals
194            bin_edges = np.round(bin_edges, 2)
195        else:
196            self._check_bin_edges(bins)
197        return bin_edges
198
199    def _check_bin_edges(self, bins):
200        """Check the bin edges."""
201        bin_edges = np.asarray(bins)
202        if np.any(bin_edges[:-1] > bin_edges[1:]):
203            raise ValueError("`bins` must increase monotonically, when an array")
